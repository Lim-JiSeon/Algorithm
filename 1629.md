```python
def draw(n, x, y):
    size = 4*n - 3
    if size == 1:
        stars[x][y] = "*"
        return
    for r in range(x, x+size):
        stars[r][y] = "*"
        stars[r][y+size-1] = "*"
    for c in range(y+1, y+size-1):
        stars[x][c] = "*"
        stars[x+size-1][c] = "*"
    draw(n-1, x+2, y+2)

n = int(input())
stars = [[" " for i in range(4*n-3)] for j in range(4*n-3)]
draw(n, 0, 0)
for s in stars:
    print("".join(s))
```
